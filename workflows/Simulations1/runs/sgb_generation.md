All MAGs generated by MEGAHIT and MetaBAT for wild_gut, cats, and dogs were copied into `mags/`.  

```
ls mags/* > /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/simulations/animal_gut/new_sgbs/mash/input_list.txt
conda activate biobakery_assembly
mkdir mash/main
mash sketch -l mash/input_list.txt -o mash/main/sketches -p 16
mash dist -p 16 -t mash/main/sketches.msh mash/main/sketches.msh > mash/mash.tsv
mkdir checkm2
cp /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/wild_gut/assembly/checkm/qa/checkm_qa_and_n50.tsv checkm2/quality_report.tsv
tail -n +2 /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/cats_out/assembly/checkm/qa/checkm_qa_and_n50.tsv >> checkm2/quality_report.tsv
tail -n +2 /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/dogs_out/assembly/checkm/qa/checkm_qa_and_n50.tsv >> checkm2/quality_report.tsv
Rscript /n/holystore01/LABS/huttenhower_lab/Users/wnickols/code/simulations/make_input_sgbs/mash_clusters.R --mash mash/mash.tsv \
  --checkm checkm2/quality_report.tsv \
  --mag_dir /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/simulations/animal_gut/new_sgbs/mags \
  --out_dir /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/simulations/animal_gut/new_sgbs/sgbs \
  --threads 16

conda deactivate
source /n/holystore01/LABS/huttenhower_lab/Users/wnickols/gtdbtk/env/bin/activate
hutlab load centos7/python3/biobakery_workflows/3.0.0-beta-devel
export GTDBTK_DATA_PATH=/n/holystore01/LABS/huttenhower_lab/Users/wnickols/code/gtdbtk/release207_v2/
conda activate gtdbtk
gtdbtk classify_wf --genome_dir /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/simulations/animal_gut/new_sgbs/sgbs/sgbs/ --out_dir /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/simulations/animal_gut/new_sgbs/gtdbtk/ -x .fa --cpus 16
python /n/holystore01/LABS/huttenhower_lab/Users/wnickols/code/simulations/gtdbtk_add_tax_assignment.py \
  --bac /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/simulations/animal_gut/new_sgbs/gtdbtk/gtdbtk.bac120.summary.tsv \
  --arc /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/simulations/animal_gut/new_sgbs/gtdbtk/gtdbtk.ar53.summary.tsv \
  --output /n/holylfs05/LABS/nguyen_lab/Everyone/wnickols/simulations/animal_gut/new_sgbs/gtdbtk/gtdbtk_relab.tsv
```
